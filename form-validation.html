表單驗證示範

請輸入您的姓名。

範例程式碼


function validateForm() {
    const nameInput = document.getElementById('name');
    const errorMessage = document.getElementById('error-message');
    const result = document.getElementById('result');

    if (nameInput.value.trim() === '') {
        errorMessage.textContent = '請輸入姓名';
        nameInput.classList.add('is-invalid'); // 添加紅色邊框
        result.textContent = ''; // 清空結果
        return false; // 阻止表單提交
    } else {
        nameInput.classList.remove('is-invalid'); // 移除紅色邊框
        result.textContent = `提交成功，姓名為: ${nameInput.value}`;
        
        // 延遲清空結果
        setTimeout(() => {
            result.textContent = ''; // 清空結果
        }, 3000); // 3000毫秒（3秒）

        return false; // 阻止表單提交
    }
}
