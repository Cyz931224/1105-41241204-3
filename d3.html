數據可視化

範例程式碼


const data = [30, 86, 168, 234, 100, 300];
const svg = d3.select('svg');

svg.selectAll('rect')
    .data(data)
    .enter()
    .append('rect')
    .attr('width', 40)
    .attr('height', d => d)
    .attr('x', (d, i) => i * 45)
    .attr('y', d => 200 - d) // 反轉 Y 軸
    .attr('fill', 'blue');
